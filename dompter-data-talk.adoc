=== Exemple C++

[source, c++]
----
#include <iostream>

using namespace std;

int main()
{
    auto sum = 0;
    for (auto i = 1; i<=5;i++) {
        if (i % 2 == 1){
            sum += i*i;
        }
    }
    cout << "Sum : " << sum;

    return 0;
}
----

=== Exemple Scala

[source, scala]
----
val pow2 = (x: Int) => x * x
val functionalResult = (1 to 5).filter(_ % 2 == 1).map(pow2).sum
println("Sum : "+functionalResult)
----
